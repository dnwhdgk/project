CREATE TABLE USERS(
    user_id     VARCHAR(20) PRIMARY KEY,
    user_pw     VARCHAR(20),
    user_name   VARCHAR(20),
    user_gender VARCHAR(4),
    user_email  VARCHAR(200),
    user_tel INT
);

CREATE TABLE AUTHMAIL(
	user_mail VARCHAR(200) PRIMARY KEY,
    auth_num INT
);

CREATE TABLE REPLY (
    reply_id    INT PRIMARY KEY AUTO_INCREMENT,
    contents    TEXT,
    reg_date    DATE,
	guest_id  INT,
    user_id INT,
    likey_id INT
);

CREATE TABLE GUEST(
guest_id INT PRIMARY KEY,
guest_num INT,
guest_type VARCHAR(20),
guest_name VARCHAR(50),
guest_img TEXT,
guest_peo INT
);

CREATE TABLE PRICE(
price_id INT PRIMARY KEY,
price_pri INT,
price_st DATE,
price_end DATE,
guest_id INT
);

CREATE TABLE GUEST_T(
guest_t_id INT PRIMARY KEY,
guest_t_name VARCHAR(50),
guest_t_add TEXT,
guest_t_img TEXT,
guest_id VARCHAR(20)
);

CREATE TABLE RESERVATION(
resev_id INT PRIMARY KEY,
resev_status VARCHAR(30),
resev_st DATE,
resev_end DATE,
user_id VARCHAR(20),
price_id INT,
guest_id INT
);

CREATE TABLE LIKEY (
likey_id INT PRIMARY KEY,
guest_id INT,
user_id VARCHAR(20)
);

CREATE TABLE PAY(
pay_id INT PRIMARY KEY,
pay_yn VARCHAR(50),
pay_kind VARCHAR(50),
pay_bank VARCHAR(50),
pay_approval INT,
pay_date DATE,
resev_id INT
);
